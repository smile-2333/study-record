function subscribe(){
  wx.requestSubscribeMessage({
    tmplIds: ['3AfajXU2xdUupGcQPJgMpg4hkX-y8Ce_iar_w5Dq6Ew'],
    success(res) {
      if (res['3AfajXU2xdUupGcQPJgMpg4hkX-y8Ce_iar_w5Dq6Ew'] === 'accept') {
        wx.showToast({
          title: '订阅OK！',
          duration: 1000,
          success(data) {
            console.log('订阅成功')
            //成功
            resolve()
          }
        })
      }
    }
  })
}

// 向外暴露接口
module.exports = {
  subscribe: subscribe
}
