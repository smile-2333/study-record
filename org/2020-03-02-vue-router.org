** 路由跳转
*** 传参
**** url后的参数
path: url?[paramName=10]
获取: this.$route.query.[paramName]
**** rest类型
path: url/:paramName
获取: this.$route.params.[paramName]
*** 跳转方式
**** 1
#+BEGIN_SRC javascript
// 传参
this.$router.push({
  path: `/describe/${id}`,
})

// 取参
this.$route.params.id
#+END_SRC
**** 2
#+BEGIN_SRC javascript
this.$router.push({
   name: 'Describe',
   params: {
   id: id
   }
})

//取参
this.$route.params.id
#+END_SRC
**** 3
#+BEGIN_SRC javascript
this.$router.push({
  path: '/describe',
  query: {
    id: id
  }
})

this.$route.query.id
#+END_SRC
